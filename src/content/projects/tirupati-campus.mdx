<script is:inline>{`
    // Set global pause state immediately
    window.pdfAutoplayPaused = true;

    document.addEventListener('DOMContentLoaded', () => {
        const btn = document.getElementById('global-pause-btn');
        const footer = document.querySelector('.site-footer');
        // Start in PAUSED state
        let paused = true;
        
        // Update button UI to show Play icon
        if (btn) {
            btn.classList.add('playing');
            btn.setAttribute('aria-label', 'Play autoplay');

            btn.addEventListener('click', () => {
                paused = !paused;
                window.pdfAutoplayPaused = paused;
                btn.classList.toggle('playing', paused);
                btn.setAttribute('aria-label', paused ? 'Play autoplay' : 'Pause autoplay');
                window.dispatchEvent(new CustomEvent('pdfAutoplayToggle', { detail: { paused } }));
            });
        }

        // Dispatch initial event to ensure all components are synced
        window.dispatchEvent(new CustomEvent('pdfAutoplayToggle', { detail: { paused } }));


        // Keep button above footer when scrolled to bottom
        function updateButtonPosition() {
            if (!footer) return;
            const footerRect = footer.getBoundingClientRect();
            const windowHeight = window.innerHeight;
            const btnHeight = btn.offsetHeight;
            const margin = 10;
            
            if (footerRect.top < windowHeight) {
                const offset = windowHeight - footerRect.top + margin;
                btn.style.bottom = offset + 'px';
            } else {
                btn.style.bottom = '1rem';
            }
        }

        window.addEventListener('scroll', updateButtonPosition);
        window.addEventListener('resize', updateButtonPosition);
        updateButtonPosition();
    });
`}</script>
